This is the APT repository for ROSE Link - VPN Router for Raspberry Pi.

Quick Install

curl -fsSL https://oussrh.github.io/ROSE-LINK/install.sh | sudo bash

Manual Setup

# Add repository
echo "deb [signed-by=/usr/share/keyrings/rose-link.gpg] https://oussrh.github.io/ROSE-LINK stable main" | sudo tee /etc/apt/sources.list.d/rose-link.list

# Update and install
sudo apt update
sudo apt install rose-link
